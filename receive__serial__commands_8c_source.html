17int ParseCommand(const char *command, int *command_id, int *command_arg) {
18 // Validate required input
19 if (!command) return 0;
20
21 // Initialize outputs to default values
22 if (command_id) *command_id = 0;
23 if (command_arg) *command_arg = 0;
24
25 // Locate the start of the angle-bracketed section
26 const char *start = strchr(command, '<');
27 if (!start) return 0;
28 start++;
29
30 // Locate the closing bracket
31 const char *end = strchr(start, '>');
32 if (!end || end <= start) return 0;
33
34 // Copy the contents between < and > into a temporary buffer
35 char buffer[64];
36 size_t len = (size_t)(end - start);
37 if (len >= sizeof(buffer)) len = sizeof(buffer) - 1;
38 strncpy(buffer, start, len);
39 buffer[len] = '\0';
40
41 int parsed = 0;
42
43 // Extract the first token (command ID)
44 char *token = strtok(buffer, ":");
45 if (!token) return 0;
46
47 // Trim leading spaces and parse command ID
48 while (isspace((unsigned char)*token)) token++;
49 if (command_id) {
50 *command_id = atoi(token);
51 parsed = 1;
52 }
53
54 // Try to parse the second token (command argument)
55 token = strtok(NULL, ":");
56 if (token && command_arg) {
57 while (isspace((unsigned char)*token)) token++;
58 *command_arg = atoi(token);
59 parsed = 2;
60 }
61
62 return parsed;
63}
