47HAL_StatusTypeDef FLASH_HELPER_Write(uint32_t address, uint8_t *data, uint32_t size) {
48 HAL_StatusTypeDef status = HAL_OK;
49 uint32_t *src = (uint32_t*)data;
50 uint32_t *dst = (uint32_t*)address;
51 uint32_t words = (size + 3) / 4; // Convert bytes to words (rounding up)
52
53 // Check if address and size are properly aligned (H7 requirements may differ)
54 #if defined(STM32H7)
55 if ((address % 32) != 0 || (size % 32) != 0) { // H7 uses 256-bit flash words (32 bytes)
56 return HAL_ERROR;
57 }
58 #else
59 if ((address % 8) != 0 || (size % 8) != 0) { // Standard STM32 64-bit requirement
60 return HAL_ERROR;
61 }
62 #endif
63
64 for (uint32_t i = 0; i < words; ) {
65 #if defined(STM32H7)
66 // H7 uses FLASHWORD programming (256 bits/32 bytes)
67 status = HAL_FLASH_Program(FLASH_TYPEPROGRAM_FLASHWORD, (uint32_t)&dst[i], (uint32_t)&src[i]);
68 i += 8; // Advance 8 words (32 bytes)
69 #else
70 // Standard double word (64-bit) programming
71 uint64_t value = ((uint64_t)src[i+1] << 32) | src[i];
72 status = HAL_FLASH_Program(FLASH_TYPEPROGRAM_DOUBLEWORD, (uint32_t)&dst[i], value);
73 i += 2; // Advance 2 words (8 bytes)
74 #endif
75
76 if (status != HAL_OK) {
77 break;
78 }
79 }
80
81 return status;
82}
