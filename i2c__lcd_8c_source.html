14void lcd_send_cmd(I2C_LCD_HandleTypeDef *lcd, char cmd)
15{
16 char upper_nibble, lower_nibble;
17 uint8_t data_t[4];
18
19 upper_nibble = (cmd & 0xF0); // Extract upper nibble
20 lower_nibble = ((cmd << 4) & 0xF0); // Extract lower nibble
21
22 data_t[0] = upper_nibble | 0x0C; // en=1, rs=0
23 data_t[1] = upper_nibble | 0x08; // en=0, rs=0
24 data_t[2] = lower_nibble | 0x0C; // en=1, rs=0
25 data_t[3] = lower_nibble | 0x08; // en=0, rs=0
26
27 HAL_I2C_Master_Transmit(lcd->hi2c, lcd->address, data_t, 4, 100);
28}

57void lcd_clear(I2C_LCD_HandleTypeDef *lcd)
58{
59 lcd_send_cmd(lcd, 0x80); // Move cursor to the home position
60 // Clear all characters
61 // 16x4 = 64 characters
62 // 20x4 = 80 characters
63 // So 80 character clearing is enough for both 16x2, 16x4, 20x2 and 20x4 displays
64 for (int i = 0; i < 80; i++)
65 {
66 lcd_send_data(lcd, ' '); // Write a space on each position
67 }
68}

77void lcd_gotoxy(I2C_LCD_HandleTypeDef *lcd, int col, int row)
78{
79 uint8_t address;
80
81 switch (row)
82 {
83 case 0: address = 0x80 + col; break; // First row
84 case 1: address = 0xC0 + col; break; // Second row
85 case 2: address = 0x94 + col; break; // Third row
86 case 3: address = 0xD4 + col; break; // Fourth row
87 default: return; // Ignore invalid row numbers
88 }
89
90 lcd_send_cmd(lcd, address); // Send command to move the cursor
91}

98void lcd_init(I2C_LCD_HandleTypeDef *lcd)
99{
100 HAL_Delay(50); // Wait for LCD power-up
101 lcd_send_cmd(lcd, 0x30); // Wake up command
102 HAL_Delay(5);
103 lcd_send_cmd(lcd, 0x30); // Wake up command
104 HAL_Delay(1);
105 lcd_send_cmd(lcd, 0x30); // Wake up command
106 HAL_Delay(10);
107 lcd_send_cmd(lcd, 0x20); // Set to 4-bit mode
108 HAL_Delay(10);
109
110 // LCD configuration commands
111 lcd_send_cmd(lcd, 0x28); // 4-bit mode, 2 lines, 5x8 font
112 HAL_Delay(1);
113 lcd_send_cmd(lcd, 0x08); // Display off, cursor off, blink off
114 HAL_Delay(1);
115 lcd_send_cmd(lcd, 0x01); // Clear display
116 HAL_Delay(2);
117 lcd_send_cmd(lcd, 0x06); // Entry mode: cursor moves right
118 HAL_Delay(1);
119 lcd_send_cmd(lcd, 0x0C); // Display on, cursor off, blink off
120}
